# 自定义样式问题

  • 0.x版本 可以通过.vuepress/override.styl文件 调整基础css样式
$accentColor = #111 //主题颜色
$textColor = #999 //字体颜色
$borderColor = #eaecef //边框颜色
$codeBgColor = #282c34 //背景颜色
